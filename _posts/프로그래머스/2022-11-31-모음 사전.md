#include <vector>

using namespace std;

int solution(string word) {
    int idx, cur, answer = 0;
    const int weights[] = { 781, 156, 31, 6, 1 };
    
    for (idx = 0; idx < word.size(); ++idx) 
    {
        switch (word[idx])
        {
            case 'A':
                cur = 0;
                break;
            case 'E':
                cur = 1;
                break;
            case 'I':
                cur = 2;
                break;
            case 'O':
                cur = 3;
                break;
            case 'U':
                cur = 4;
                break;
        }
        ++answer;
        answer += cur * weights[idx];
    }
    return answer;
}
